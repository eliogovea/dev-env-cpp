1 /*
2  * The program executes a command in a child process, redirects its output and
3  * error to pipes, and handles timeouts and signals in the parent process.
4  *
5  * Command-line arguments:
6  * - [--timeout|-t TIMEOUT]: Optional flag to specify a timeout in seconds.
7  * Defaults to 10 seconds if not provided.
8  * - COMMAND [ARGS...]: The command to execute and its arguments. This must be
9  * provided.
10  *
11  * The program will:
12  * 1. Set up pipes for stdout and stderr.
13  * 2. Fork a child process.
14  * 3. In the child process, redirect stdout and stderr to the pipes and execute the command.
15  * 4. In the parent process, use `poll` to monitor pipes and handle timeouts and signals.
16  * 5. Terminate the child process if it does not complete within the specified timeout.
17  */

48 // Function to create and configure a timerfd
49 int create_timer_fd(int timeout)
50 {
51  // Create a timer file descriptor with CLOCK_MONOTONIC and non-blocking mode
52  // CLOCK_MONOTONIC ensures that the timer is not affected by system clock changes
53  int timer_fd = timerfd_create(CLOCK_MONOTONIC, TFD_NONBLOCK);
54  if (timer_fd == -1) {
55  perror("timerfd_create failed");
56  exit(EXIT_FAILURE);
57  }
58 
59  // Configure the timer to expire after the specified timeout
60  struct itimerspec timer_spec;
61  timer_spec.it_interval.tv_sec = 0; // No periodic timer
62  timer_spec.it_interval.tv_nsec = 0; //
63  timer_spec.it_value.tv_sec = timeout; // Set initial expiration time
64  timer_spec.it_value.tv_nsec = 0; //
65 
66  // Set the timer with the specified timeout value
67  if (timerfd_settime(timer_fd, 0, &timer_spec, NULL) == -1) {
68  perror("timerfd_settime failed");
69  close(timer_fd);
70  exit(EXIT_FAILURE);
71  }
72 
73  return timer_fd;
74 }
